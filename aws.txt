# Install Docker on AWS EC2 Instance
sudo kill -9 1962
sudo killall apt apt-get
sudo apt-get update
sudo apt-get install -y apt-transport-https ca-certificates curl software-properties-common
curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo apt-key add -
sudo add-apt-repository \
   "deb [arch=amd64] https://download.docker.com/linux/ubuntu \
   $(lsb_release -cs) \
   stable"
sudo apt-get update
sudo apt-get install -y docker-ce
sudo systemctl start docker
sudo systemctl enable docker
sudo usermod -aG docker $USER

# Verify Docker installation
docker --version

# Grant Permissions and Restart Docker
sudo usermod -aG docker $USER
groups $USER
sudo systemctl restart docker
ls -l /var/run/docker.sock
sudo chmod 666 /var/run/docker.sock
groups $USER
sudo systemctl restart docker

# Log in to Docker on the EC2 Instance

# Create a Network on AWS
docker network create my-network


# Run Web Application Container

docker run -d --name webapp --network my-network -p 8080:8080 -e JAVA_OPTS="-Denv=docker" chaitanyabhatt147/docker-04-webapp:latest 

# Add data in data base
docker exec -it mysql mysql -u root -p
